1. send otp for set password :


endpoint : "https://yus.kwscloud.in/yus/send-otp-driver-password"
method : Post (body -> x-www-form-urlencoded) 

frontend -> backend(request):
{
    "driver_id" : 1001,
    "email"     : "driver@gmail.com"
}

backend -> frontend (response):
{
    "status": "no driver found" (or)
    "otp_sent" : true (or)
    "otp_sent" : false
}
